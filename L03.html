let tydzien = ["Poniedziałek", "Wtorek", "Środa", "Czwartek", "Piątek", "Sobota", "Niedziela"]

tydzien[2] // "Środa"

tydzien[2][0] // "Ś"

tydzien[2][3] // "d"

tydzien[2] = "dupa" // "dupa" wchodzi w miejsce "Środa"

tydzien // ["Poniedziałek", "Wtorek", "dupa" , "Czwartek", "Piątek", "Sobota", "Niedziela"]

tydzien[10] = "jopjopjop" // ["Poniedziałek", "Wtorek", "dupa", "Czwartek", "Piątek", "Sobota", "Niedziela", undefined, undefined, undefined, "jopjopjop"]

2. Array Methods

- Push // adds a position to the end

- Pop // removes a position from the end

- Shift // removes a position from the start

- Unshift // adds a position to the start

tydzien.push("8smy dzien tygodnia") // ["Poniedziałek", "Wtorek", "dupa", "Czwartek", "Piątek", "Sobota", "Niedziela", undefined, undefined, undefined, "jopjopjop", "8smy dzien tygodnia"]

tydzien.pop() // odejmie ostatni wpis z arraya > ["Poniedziałek", "Wtorek", "dupa", "Czwartek", "Piątek", "Sobota", "Niedziela", undefined, undefined, undefined, "jopjopjop"]

let zapisanyPop = tydzien.pop() // "jopjopjop" > ["Poniedziałek", "Wtorek", "dupa", "Czwartek", "Piątek", "Sobota", "Niedziela", undefined, undefined, undefined]

let rozwalTydzien = tydzien.shift() // "Poniedziałek" > zasada działania jak w przypadku pop

tydzien.unshift("Poniedziałek") zasada jak w przypadku push, tylko na start

- concat // merge arrays

- includes // look for a value

- indexOf // just like string.indexOf

- join // creates a string from an array

- reverse // reverse an array

- slice // copies a portion on an array

- splice // removes/replaces element

- sort // sorts an array

let dniPracujace = ["Poniedziałek", "Wtorek", "Środa", "Czwartek", "Piątek"]

let dniWeekendowe = ["Sobota", "Niedziela"]

let pelnyTydzien = dniPracujace.concat(dniWeekendowe);

pelnyTydzien.includes("Sobota") // true

pelnyTydzien.indexOf("Sobota") // 5

dniWeekendowe.reverse() // ["Niedziela", "Sobota"]
